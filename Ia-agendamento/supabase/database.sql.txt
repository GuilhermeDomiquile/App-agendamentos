create table public.agendamentos (
  id bigserial not null,
  telefone text null,
  nome_cliente text null,
  servico text null,
  data date null,
  hora time without time zone null,
  reservado_ate timestamp with time zone null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  status text null,
  uuid uuid null,
  id_calendar text null,
  constraint agendamentos_pkey primary key (id)
) TABLESPACE pg_default;

create trigger trigger_bloquear_horario_ocupado BEFORE INSERT
or
update on agendamentos for EACH row
execute FUNCTION bloquear_horario_ocupado ();

create trigger trigger_validar_horario_agendamento BEFORE INSERT
or
update on agendamentos for EACH row
execute FUNCTION validar_horario_agendamento ();

create trigger trigger_validar_intervalo_30_min BEFORE INSERT
or
update on agendamentos for EACH row
execute FUNCTION validar_intervalo_30_min ();

create table public.bloqueios_agenda (
  id uuid not null default gen_random_uuid (),
  tipo text not null,
  dia_semana integer null,
  data date null,
  hora_inicio time without time zone null,
  hora_fim time without time zone null,
  motivo text null,
  ativo boolean null default true,
  created_at timestamp with time zone null default now(),
  constraint bloqueios_agenda_pkey primary key (id),
  constraint bloqueios_agenda_tipo_check check (
    (
      tipo = any (array['recorrente'::text, 'data'::text])
    )
  )
) TABLESPACE pg_default;


----------------------------------------------------


create table public.configuracao_agenda (
  id uuid not null default extensions.uuid_generate_v4 (),
  hora_inicio time without time zone not null default '06:00:00'::time without time zone,
  hora_fim time without time zone not null default '18:00:00'::time without time zone,
  dias_permitidos integer[] not null default array[1, 2, 3, 4, 5, 6],
  constraint configuracao_agenda_pkey primary key (id)
) TABLESPACE pg_default;

create table public.dados_clientes (
  id bigserial not null,
  created_at timestamp with time zone null,
  telefone text null,
  nomewpp text null,
  em_atendimento boolean not null default true,
  ticket_id text null default ''::text,
  atendimento_uuid uuid null default extensions.uuid_generate_v4 (),
  constraint dados_cliente_pkey primary key (id)
) TABLESPACE pg_default;

create trigger trigger_novo_atendimento_uuid BEFORE
update on dados_clientes for EACH row
execute FUNCTION gerar_uuid_novo_atendimento ();

create table public.notifica_agendamento (

create table public.servicos (
  id bigint generated by default as identity not null,
  nome text not null,
  preco numeric(10, 2) not null,
  ativo boolean null default true,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  ordem integer null,
  constraint servicos_pkey primary key (id)
) TABLESPACE pg_default;
